191   private def matchParts(uri: String): (Option[String], Option[String], Option[String], Option[String], Option[String]) = {
192 
193     val chars = uri.toCharArray
194     val length = chars.length
195     var i = 0
196 
197     def readUntil(delimiters: Set[Char]): String = {
198       if (i >= length) ""
199       else {
200         val res = StringBuilder.newBuilder
201         while (i < length && !delimiters.contains(chars(i))) {
202           res.append(chars(i))
203           i += 1
204         }
205         res.toString()
206       }
207     }
208 
209     def read(): String = {
210       if (i >= length) ""
211       else {
212         val res = StringBuilder.newBuilder
213         while (i < length) {
214           res.append(i)
215           i += 1
216         }
217         res.toString()
218       }
219     }
220 
221     var scheme: Option[String] = Some(readUntil(Set(':', '/', '?', '#')))
222     if (i < length && chars(i) == ':') {
223       i += 1
224     } else {
225       scheme = None
226       i = 0
227     }
228     val authority = {
229       if (i + 1 < length && chars(i) == '/' && chars(i + 1) == '/') {
230         i = i + 2
231         Some(readUntil(Set('/', '?', '#')))
232       } else {
233         None
234       }
235     }
236     val path = if (i < length && !Set('?', '#').contains(chars(i))) {
237       Some(readUntil(Set('?', '#')))
238     } else {
239       None
240     }
241 
242     val query = if (i < length && chars(i) == '?') {
243       i += 1
244       Some(readUntil(Set('#')))
245     } else {
246       None
247     }
248 
249     val fragment = if (i < length && chars(i) == '#') {
250       i += 1
251       Some(read())
252     } else {
253       None
254     }
255 
256     (scheme, authority, path, query, fragment)
257   }

259   def parse(value: String): Either[UriParseError, Uri] = {
260 
261     matchParts(value) match {
262       case (s, a, p, q, f) =>
263         println(s"s = '$s', a = '$a', p = '$p', q = '$q', f = '$f'")
264         val scheme: Either[SchemeError, Scheme] = notNull(MissingScheme(value), s).flatMap(Scheme.apply)
265         val authority: Either[AuthorityParseError, Option[Authority]] = sequence(a.map(Authority.parse))
266         (scheme, authority) match {
267           case (Right(s), Right(a)) =>
268             val path = Path.parse(p.getOrElse(""))
269             val query = q.map(q => Query(Encoder.decode(q)))
270             val fragment = f.map(f => Fragment(Encoder.decode(f)))
271             // The regular expression makes it impossible for a PathError to occur
272             Right(Uri.of(s, a, path, query, fragment).right.get)
273           case _ =>
274             Left(UriParseError(scheme.left.toOption, authority.left.toOption))
275         }
276       case _ =>
277         // It actually never happens. Every possible string matches de url regexp, afaik
278         throw new IllegalStateException("Unreachable code")
279     }
