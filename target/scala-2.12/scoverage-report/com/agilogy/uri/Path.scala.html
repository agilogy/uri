26 object Path {
27   val empty: EmptyPath.type = EmptyPath
28   val Slash: AbsoluteSingleSegmentPath = Path / ""
29   def /(s: Segment): AbsoluteSingleSegmentPath = AbsoluteSingleSegmentPath(s)
30   def /(s: String): AbsoluteSingleSegmentPath = Path / Segment(s)
31   def apply(s: NonEmptySegment) = RootlessSingleSegmentPath(s)
32   def apply(s: String): PathRootlessEmpty = Segment(s) match {
33     case EmptySegment       => Path.empty
34     case s: NonEmptySegment => RootlessSingleSegmentPath(s)
35   }
36 
37   def absoluteOrEmpty(s: Segment*): PathAbEmpty = {
38     s match {
39       case Seq()     => empty
40       case h +: tail => absolute(h, tail: _*)
41     }
42   }
43   def absolute(segment: Segment, moreSegments: Segment*): AbsolutePath = AbsolutePath(segment, moreSegments: _*)
44   def rootless(segment: NonEmptySegment, moreSegments: Segment*): RootlessPath = RootlessPath(segment, moreSegments: _*)
45 
46   private val AbsolutePathRe = "(/[^?#]*)+".r
47   private val RelativePathRe = "([^?#/]+)(/[^?#]*)*".r
48 
49   private val SegmentRe = "/([^?#/]*)".r
50 
51   def parse(s: String): Path = {
52     s match {
53       case "" => Path.empty
54       case AbsolutePathRe(_) =>
55         val res = SegmentRe.findAllIn(s).matchData.foldLeft[PathAbEmpty](Path.empty) {
56           case (acc, m) => acc / Encoder.decode(m.group(1))
57         }
58         res
59       case RelativePathRe(headSegment, segmentsOrNull) =>
60         val segmentsString = Option(segmentsOrNull).getOrElse("")
61         val tailSegments = SegmentRe.findAllIn(segmentsString).matchData.map(m => m.group(1))
62         val h = NonEmptySegment(Encoder.decode(headSegment))
63         val t = tailSegments.toList.map(s => Segment(Encoder.decode(s)))
64         Path.rootless(h, t: _*)
65     }
66   }
