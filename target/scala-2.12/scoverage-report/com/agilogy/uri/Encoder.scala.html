144   def quoteUri(uri: Uri): String = {
145     val sScheme = uri.scheme.stringValue
146     val sAuthority = uri.authority.map(a => "//" + a.stringValue).getOrElse("")
147     val sPath = uri.path.stringValue
148     val sQuery = uri.query.map(q => "?" + Encoder.quote(q.stringValue, Encoder.queryChars)).getOrElse("")
149     val sFragment = uri.fragment.map(f => "#" + f.stringValue).getOrElse("")
150     s"$sScheme:$sAuthority$sPath$sQuery$sFragment"
151   }
